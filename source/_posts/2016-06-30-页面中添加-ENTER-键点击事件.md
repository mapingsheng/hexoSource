title: 页面中添加'ENTER'键点击事件
date: 2016-06-30 16:31:17
categories: javascript
tags: 添加enter事件
---
# jquery #
	$("body").keydown(function(e){
        if(e.keyCode=="13"){
            alert('点击了Enter键');
        }
    });
# javascript #
	document.body.onclick=function(){
        if(e.keyCode=="13"){
           alert('点击了Enter键');
        }
    };
